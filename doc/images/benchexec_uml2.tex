          %font = \footnotesize,

    \umlaggreg[geometry=--]{RunSet}{SourceFileSet};
    \umlaggreg[geometry=--, arg2=1..n]{RunSet}{Run};
    \umlaggreg[geometry=-|-, anchor2=170, arg2=1..n, pos2=2.6]{SourceFileSet}{desc};
